Seamless Integration

Connect to the lightweight Python production API.

request.py
import requests

url = "https://nara-t.api/recommend"
payload = {
    "user_id": "usr_9x2L",
    "session_id": "sess_1a",
    "context": {
        "hour": 22.5,
        "device": "speaker",
        "session_phase": "winddown",
        "is_sleep": True
    },
    "num_results": 10
}

response = requests.post(url, json=payload)
print(response.json())
